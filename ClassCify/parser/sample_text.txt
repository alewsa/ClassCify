(class MyClass
  ((vardec Int x)
   (vardec Boolean flag))
  (init ()
    (super))
  (method doIt () Void
    (vardec Int y)
    (= y (+ 
            (+ (- 10 5)
               (* 2 3))
            (/ 8 2)))
    (if (< y 20)
      (return (println y)))
    (return)))
(vardec MyClass obj)
(= obj (new MyClass))
(call obj doIt)
